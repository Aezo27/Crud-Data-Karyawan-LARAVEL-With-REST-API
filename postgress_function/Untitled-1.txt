CREATE OR REPLACE FUNCTION "public"."select_data"("_id" int4)
  RETURNS TABLE("nama" varchar, "nip" varchar, "alamat" varchar, "jenis_kelamin" varchar, "divisi" varchar) AS $BODY$
BEGIN
    RETURN QUERY SELECT
        data_pegawais.nama,
        data_pegawais.nip,
				data_pegawais.alamat,
				data_pegawais.jenis_kelamin,
				data_pegawais.divisi
    FROM
        data_pegawais
    WHERE
        "id" = _id ;
END; $BODY$
  LANGUAGE plpgsql VOLATILE
  COST 100
  ROWS 1000
